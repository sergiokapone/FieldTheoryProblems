\begin{problem}
    –ù–∞ –ø–ª–æ—Å–∫—ñ–π –ø–æ–≤–µ—Ä—Ö–Ω—ñ, —â–æ —Ä–æ–∑–¥—ñ–ª—è—î —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞ $1$ —Ç–∞ $2$, –∑–∞–¥–∞–Ω–∞ –ø–æ–≤–µ—Ä—Ö–Ω–µ–≤–∞ –≥—É—Å—Ç–∏–Ω–∞ –≤—ñ–ª—å–Ω–æ–≥–æ –∑–∞—Ä—è–¥—É $\sigma$. –£ —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞—Ö –¥—ñ–µ–ª–µ–∫—Ç—Ä–∏—á–Ω—ñ –ø—Ä–æ–Ω–∏–∫–Ω–æ—Å—Ç—ñ, –≤—ñ–¥–ø–æ–≤—ñ–¥–Ω–æ, $\epsilon_1 > 1$ —Ç–∞ $\epsilon_2 > 1$ . –ù–æ—Ä–º–∞–ª—å $\vect{n}$ –¥–æ –ø–æ–≤–µ—Ä—Ö–Ω—ñ —Å–ø—Ä—è–º–æ–≤–∞–Ω–∞ –∑ —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞ $2$ –≤ $1$. –£ —Å–µ—Ä–µ–¥–æ–≤–∏—â—ñ $1$ –∑–∞–¥–∞–Ω–∞ –Ω–∞–ø—Ä—É–∂–µ–Ω—ñ—Å—Ç—å –µ–ª–µ–∫—Ç—Ä–∏—á–Ω–æ–≥–æ –ø–æ–ª—è $\Efield_1$. –ó–Ω–∞–π—Ç–∏ –Ω–∞–ø—Ä—É–∂–µ–Ω—ñ—Å—Ç—å –µ–ª–µ–∫—Ç—Ä–∏—á–Ω–æ–≥–æ –ø–æ–ª—è –≤ —Å–µ—Ä–µ–¥–æ–≤–∏—â—ñ $2$ —É –≤–µ–∫—Ç–æ—Ä–Ω–æ–º—É –≤–∏–≥–ª—è–¥—ñ.
\begin{solution}
    $\Efield_2 = \Efield_1 + \left( \frac{\epsilon_1}{\epsilon _2} - 1 \right)\vect{n}\left( \vect{n} \cdot \Efield_1 \right) - \frac{4\pi \sigma }{\epsilon _2}{\vect{n}}$.
\end{solution}
\end{problem}

\begin{problem}
–¢–æ—á–∫–æ–≤–∏–π –∑–∞—Ä—è–¥ $q$ —Ä–æ–∑–º—ñ—â–µ–Ω–∏–π –Ω–∞ –ø–ª–æ—Å–∫—ñ–π –≥—Ä–∞–Ω–∏—Ü—ñ —Ä–æ–∑–¥—ñ–ª—É –¥–≤–æ—Ö –æ–¥–Ω–æ—Ä—ñ–¥–Ω–∏—Ö  –¥—ñ–µ–ª–µ–∫—Ç—Ä–∏–∫—ñ–≤ (–≥—Ä–∞–Ω–∏—Ü—é —Ä–æ–∑–¥—ñ–ª—É –≤–≤–∞–∂–∞—Ç–∏ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—é) –∑ –ø—Ä–æ–Ω–∏–∫–Ω–æ—Å—Ç—è–º–∏ $\epsilon_{1}$ —ñ $\epsilon_{2}$. –ó–Ω–∞–π—Ç–∏ –Ω–∞–ø—Ä—É–∂–µ–Ω—ñ—Å—Ç—å —ñ —ñ–Ω–¥—É–∫—Ü—ñ—é –µ–ª–µ–∫—Ç—Ä–∏—á–Ω–æ–≥–æ –ø–æ–ª—è, –∞ —Ç–∞–∫–æ–∂ –π–æ–≥–æ –ø–æ—Ç–µ–Ω—Ü—ñ–∞–ª –≤ —É—Å—å–æ–º—É –ø—Ä–æ—Å—Ç–æ—Ä—ñ.
\begin{solution}
	$
		\pot  = \frac{2}{{{\varepsilon _1} + {\varepsilon _2}}}\frac{q}{r},
	$
	$
		\Efield = \frac{2q}{\epsilon_1 + \epsilon_2}\frac{\vect{r}}{r^3},
	$
	$ \Dfield_i = \frac{2\epsilon_i}{\epsilon_1 + \epsilon_2}\frac{q\vect{r}}{r^3} $ ($i = 1,2$).
\end{solution}
\end{problem}

\begin{problem}
–î—ñ–µ–ª–µ–∫—Ç—Ä–∏—á–Ω—É –∫—É–ª—é —Ä–∞–¥—ñ—É—Å–∞ $R$ (–¥—ñ–µ–ª–µ–∫—Ç—Ä–∏—á–Ω–∞ –ø—Ä–æ–Ω–∏–∫–Ω—ñ—Å—Ç—å –∫—É–ª—ñ $\epsilon^{(i)}$) –≤–Ω–æ—Å—è—Ç—å —É –∑–æ–≤–Ω—ñ—à–Ω—î –æ–¥–Ω–æ—Ä—ñ–¥–Ω–µ –µ–ª–µ–∫—Ç—Ä–∏—á–Ω–µ –ø–æ–ª–µ –∑ –Ω–∞–ø—Ä—É–∂–µ–Ω—ñ—Å—Ç—é $\Efield_0$ . –î—ñ–µ–ª–µ–∫—Ç—Ä–∏—á–Ω–∞ –ø—Ä–æ–Ω–∏–∫–Ω—ñ—Å—Ç—å —Å–µ—Ä–µ–¥–æ–≤–∏—â–∞ –Ω–∞–≤–∫–æ–ª–æ –∫—É–ª—ñ –¥–æ—Ä—ñ–≤–Ω—é—î $\epsilon^{(e)}$. –ó–Ω–∞–π—Ç–∏ –Ω–∞–ø—Ä—É–∂–µ–Ω—ñ—Å—Ç—å —ñ –ø–æ—Ç–µ–Ω—Ü—ñ–∞–ª –ø–æ–ª—è –≤ —Å–µ—Ä–µ–¥–∏–Ω—ñ —ñ –∑–æ–≤–Ω—ñ –∫—É–ª—ñ. –ó–Ω–∞–π—Ç–∏ –≤–µ–∫—Ç–æ—Ä –ø–æ–ª—è—Ä–∏–∑–∞—Ü—ñ—ó –∫—É–ª—ñ —ñ –≥—É—Å—Ç–∏–Ω—É –∑–≤'—è–∑–∞–Ω–∏—Ö –∑–∞—Ä—è–¥—ñ–≤ –Ω–∞ –π–æ–≥–æ –ø–æ–≤–µ—Ä—Ö–Ω—ñ.
\begin{solution}
	–ó–æ–≤–Ω—ñ –∫—É–ª—ñ –Ω–∞ –∑–æ–≤–Ω—ñ—à–Ω—î –æ–¥–Ω–æ—Ä—ñ–¥–Ω–µ –ø–æ–ª–µ –Ω–∞–∫–ª–∞–¥–∞—î—Ç—å—Å—è –ø–æ–ª–µ, —â–æ —Å—Ç–≤–æ—Ä—é—î—Ç—å—Å—è –ø–æ–ª—è—Ä–∏–∑–∞—Ü—ñ–π–Ω–∏–º–∏ –∑–∞—Ä—è–¥–∞–º–∏ —è–∫—ñ –≤–∏–Ω–∏–∫–ª–∏ –Ω–∞ –∫—É–ª—ñ. –ó —É—Ä–∞—Ö—É–≤–∞–Ω–Ω—è–º –Ω—É–ª—å–æ–≤–æ—ó —É–º–æ–≤–∏ –Ω–∞ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ –¥–ª—è –ø–æ–ª—è –∑–∞—Ä—è–¥—ñ–≤ –Ω–∞ –∫—É–ª—ñ —Ç–∞ —É–º–æ–≤–∏ —Ä–µ–≥—É–ª—è—Ä–Ω–æ—Å—Ç—ñ –≤ –ø–æ—á–∞—Ç–∫—É –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç, —à—É–∫–∞—Ç–∏–º–µ–º–æ –ø–æ—Ç–µ–Ω—Ü—ñ–∞–ª —É –≤–∏–≥–ª—è–¥—ñ:
	\[
		\pot(r) =
		\begin{cases}
			-Cr\cos\theta, \quad r \le R \\
			-E_0 r \cos\theta + \frac{p\cos\theta}{r^2}, \quad r > R.
		\end{cases}
	\]
	–ì—Ä–∞–Ω–∏—á–Ω—ñ —É–º–æ–≤–∏ –º–∞—é—Ç—å –≤–∏–≥–ª—è–¥:
	\[
		\left. \pot^{(i)} \right|_{r = R} = \left. \pot^{(e)} \right|_{r = R} \left., \quad \epsilon^{(i)}\frac{\partial\pot^{(i)}}{\partial r}\right|_{r = R} =  \left. \epsilon^{(e)}\frac{\partial\pot^{(e)}}{\partial r}\right|_{r = R},
	\]
	—ñ –¥–∞—é—Ç—å –∑–º–æ–≥—É –≤–∏–∑–Ω–∞—á–∏—Ç–∏ –∫–æ–Ω—Å—Ç–∞–Ω—Ç—É $C$ —Ç–∞ –¥–∏–ø–æ–ª—å–Ω–∏–π –º–æ–º–µ–Ω—Ç $p$:
	\[
		\begin{cases}
			-CR\cos\theta = -E_0 R \cos\theta + \frac{p\cos\theta}{R^2} \\
			\epsilon^{(i)}(-C\cos\theta) = \epsilon^{(e)} E_0 \cos\theta + \epsilon^{(e)}\frac{2p\cos\theta}{R^3}.
		\end{cases}
	\]
	–ó–≤—ñ–¥–∫–∏:
	\[
		p = \frac{\epsilon^{(i)} - \epsilon^{(e)}}{\epsilon^{(i)} + 2\epsilon^{(e)}}R^3E_0,
	\]
	\[
		C = \frac{3\epsilon^{(e)}}{\epsilon^{(i)} + 2\epsilon^{(e)}}.
	\]
	–ü–æ—Ç–µ–Ω—Ü—ñ–∞–ª –¥–æ—Ä—ñ–≤–Ω—é—î:
	\[
		\pot =
		\begin{cases}
			-\frac{3\epsilon^{(e)}}{\epsilon^{(i)} + 2\epsilon^{(e)}}\left( \Efield_0\cdot \vect{r}\right), & r \le R \\
			-\left( \Efield_0\cdot \vect{r}\right) + \frac{\vect{p} \vect{r}}{r^3},                         & r > R.
		\end{cases}
	\]
	–ì—É—Å—Ç–∏–Ω–∞ –∑–≤'—è–∑–∞–Ω–æ–≥–æ –∑–∞—Ä—è–¥—É:
	\[
		\sigma_   \text{–∑–≤'—è–∑} = P_{1n} - P_{2n} = \frac{1}{4\pi} \left( \left.\frac{\partial\pot^{(i)}}{\partial r}\right|_{r = R} -  \left. \frac{\partial\pot^{(e)}}{\partial r}\right|_{r = R} \right)  = \frac{3}{4\pi} \frac{\epsilon_i - \epsilon_e}{\epsilon_i + 2\epsilon_e} E_0\cos\theta.
	\]
\end{solution}
\end{problem}

	\emph{–í–∫–∞–∑—ñ–≤–∫–∞}: –®—É–∫–∞—î–º–æ —Ä–æ–∑–≤‚Äô—è–∑–æ–∫ —Ä—ñ–≤–Ω—è–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ –∫—É–ª—ñ (–≤ —Å—Ñ–µ—Ä–∏—á–Ω–∏—Ö –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞—Ö)  —É –≤–∏–≥–ª—è–¥—ñ $\pot(r,\theta,\phi) = f(r)\cos\theta$. –ú–∞—î–º–æ –∑–∞–≥–∞–ª—å–Ω–∏–π —Ä–æ–∑–≤'—è–∑–æ–∫ –¥–ª—è  $f(r) = A\frac{r}{R} + B\left( \frac{R}{r}\right)^2$.  –ö–æ–Ω—Å—Ç–∞–Ω—Ç–∏ $A$ —Ç–∞ $B$ –≤ —Å–µ—Ä–µ–¥–∏–Ω—ñ —Ç–∞ –∑–æ–≤–Ω—ñ –∫—É–ª—ñ —Ä—ñ–∑–Ω—ñ. –ó–Ω–∞—Ö–æ–¥–∏–º–æ —ñ—Ö –∑ –Ω—É–ª—å–æ–≤–æ—ó —É–º–æ–≤–∏ –Ω–∞ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ –ø—Ä–∏ $r>R$, —É–º–æ–≤–∏ —Ä–µ–≥—É–ª—è—Ä–Ω–æ—Å—Ç—ñ —Ç–∞ –∑ –≥—Ä–∞–Ω–∏—á–Ω–æ—ó —É–º–æ–≤–∏ –ø—Ä–∏  $r = R$. –ì—É—Å—Ç–∏–Ω—É –≤—ñ–ª—å–Ω–æ–≥–æ –∑–∞—Ä—è–¥—É –∑–Ω–∞—Ö–æ–¥–∏–º–æ –∑ –≥—Ä–∞–Ω–∏—á–Ω–æ—ó —É–º–æ–≤–∏ –¥–ª—è –Ω–æ—Ä–º–∞–ª—å–Ω–æ—ó –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∏ —ñ–Ω–¥—É–∫—Ü—ñ—ó –Ω–∞ –ø–æ–≤–µ—Ä—Ö–Ω—ñ –∫—É–ª—ñ.
\end{solution}
\end{problem}

	\(
		\pot(R,z,\phi) =
		\begin{cases}
			\pot _0\left( \frac{r}{R} \right)^m\cos (m\varphi), r < R \\
			\pot _0\left( \frac{R}{r} \right)^m\cos (m\varphi), r \ge R.
		\end{cases}
	\),
	\(
		\sigma_\text{–≤—ñ–ª—å–Ω} = \frac{m(\varepsilon  + 1)}{4\pi }\cos (m\varphi ).
	\)
\end{solution}
\end{problem}

\begin{problem}\label{prb:L}
–ï–ª–µ–∫—Ç—Ä–∏—á–Ω–µ –ø–æ–ª–µ —É—Ç–≤–æ—Ä–µ–Ω–µ —Ä–æ–∑–ø–æ–¥—ñ–ª–æ–º –∑–∞—Ä—è–¥—ñ–≤
\[
	\sigma(\theta, \phi) = \sum\limits_{n=0}^{\infty}\sum\limits_{m = -n}^ns_{mn}Y_{nm}(\theta, \phi)
\]
(–¥–µ –ø—Ä–∏–ø—É—Å–∫–∞—î–º–æ, —â–æ —Ä—è–¥ —à–≤–∏–¥–∫–æ –∑–±—ñ–≥–∞—î—Ç—å—Å—è)  –Ω–∞ –ø–æ–≤–µ—Ä—Ö–Ω—ñ –∫—É–ª—ñ —Ä–∞–¥—ñ—É—Å–∞ $R$, —ñ–Ω—à–∏—Ö –∑–∞—Ä—è–¥—ñ–≤ –Ω–µ–º–∞—î. –£—Å–µ—Ä–µ–¥–∏–Ω—ñ –∫—É–ª—ñ –¥—ñ–µ–ª–µ–∫—Ç—Ä–∏—á–Ω–∞ –ø—Ä–æ–Ω–∏–∫–Ω—ñ—Å—Ç—å $\epsilon$, –∑–æ–≤–Ω—ñ~--- –≤–∞–∫—É—É–º. –ó–Ω–∞–π—Ç–∏ –ø–æ—Ç–µ–Ω—Ü—ñ–∞–ª  $\pot(r,\theta,\phi)$ (–∑–æ–≤–Ω—ñ —Ç–∞ —É—Å–µ—Ä–µ–¥–∏–Ω—ñ –∫—É–ª—ñ, –≤ —Å—Ñ–µ—Ä–∏—á–Ω–∏—Ö –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞—Ö  $r$, $\theta$, $\phi$).
\begin{solution}
	–†–æ–∑–≤'—è–∑–æ–∫ —Å–ª—ñ–¥ —à—É–∫–∞—Ç–∏ –æ–∫—Ä–µ–º–æ –∑–æ–≤–Ω—ñ –∫—É–ª—ñ ($r >R$ ) —Ç–∞ –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ ($r<R$). –ö–æ—Ä–∏—Å—Ç—É—é—á–∏—Å—å –ø–æ–≤–Ω–æ—Ç–æ—é —Å–∏—Å—Ç–µ–º–∏ —Å—Ñ–µ—Ä–∏—á–Ω–∏—Ö —Ñ—É–Ω–∫—Ü—ñ–π –Ω–∞ —Å—Ñ–µ—Ä—ñ, —à—É–∫–∞—î–º–æ —Ä–æ–∑–≤‚Äô—è–∑–æ–∫ —É –≤–∏–≥–ª—è–¥—ñ:
	\[
		\pot(r,\theta ,\varphi ) = \sum\limits_{n = 0}^\infty  {\sum\limits_{m =  - n}^n {{f_{nm}}(r){Y_{nm}}(\theta ,\varphi )} } .
	\]
	–ü—ñ–¥—Å—Ç–∞–≤–∏–º–æ —Ü–µ –≤ —Ä—ñ–≤–Ω—è–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ –≤ —Å—Ñ–µ—Ä–∏—á–Ω–∏—Ö –∫–æ–æ—Ä–¥–∏–Ω–∞—Ç–∞—Ö:
	\begin{equation}\label{Laplace_prb:L}
		\Delta \pot  \equiv \frac{1}{{{r^2}}}\frac{\partial }{{\partial r}}\left( {{r^2}\frac{{\partial \pot }}{{\partial r}}} \right) + \frac{1}{{{r^2}}}\hat \Lambda \psi  = 0,
	\end{equation}
	–¥–µ \(\hat \Lambda {Y_{nm}} \equiv \frac{1}{{\sin \theta }}\frac{\partial }{{\partial \theta }}\left( {\sin \theta \frac{{\partial {Y_{nm}}}}{{\partial \theta }}} \right) + \frac{1}{{{{\sin }^2}\theta }}\frac{{{\partial ^2}{Y_{nm}}}}{{\partial {\varphi ^2}}} =  - n\left( {n + 1} \right){Y_{nm}}\left( {\theta ,\varphi } \right)\) .

	–ó–≤—ñ–¥—Å–∏, –∑ –æ–≥–ª—è–¥—É –Ω–∞ –Ω–µ–∑–∞–ª–µ–∂–Ω—ñ—Å—Ç—å —Å—Ñ–µ—Ä–∏—á–Ω–∏—Ö —Ñ—É–Ω–∫—Ü—ñ–π –∑ —Ä—ñ–∑–Ω–∏–º–∏ —ñ–Ω–¥–µ–∫—Å–∞–º–∏ —ñ –ª—ñ–Ω—ñ–π–Ω—ñ—Å—Ç—å —Ä—ñ–≤–Ω—è–Ω–Ω—è –õ–∞–ø–ª–∞—Å–∞ –ø–æ $\pot$  (–ø—Ä–∏–Ω—Ü–∏–ø —Å—É–ø–µ—Ä–ø–æ–∑–∏—Ü—ñ—ó!), –∫–æ–∂–Ω—É —Å–∫–ª–∞–¥–æ–≤—É –ø–æ—Ç–µ–Ω—Ü—ñ–∞–ª—É –º–æ–∂–Ω–∞ —Ä–æ–∑–≥–ª—è–¥–∞—Ç–∏ –æ–∫—Ä–µ–º–æ. –ú–∞—î–º–æ
	\[
		\frac{1}{{{r^2}}}\frac{\partial }{{\partial r}}\left( {{r^2}\frac{{\partial {f_{nm}}}}{{\partial r}}} \right) - \frac{{n\left( {n + 1} \right)}}{{{r^2}}}{f_{nm}}(r) = 0
	\]
	–¥–ª—è —É—Å—ñ—Ö $n= 0,1,2,\ldots$ , —ñ $m = -n, -(n-1), \ldots, 0, \ldots, n-1, n$  –¥–ª—è –∫–æ–∂–Ω–æ–≥–æ $n$.
	–ó–∞–≥–∞–ª—å–Ω–∏–π —Ä–æ–∑–≤‚Äô—è–∑–æ–∫ —Ä—ñ–≤–Ω—è–Ω–Ω—è~\eqref{Laplace_prb:L} —î –∫–æ–º–±—ñ–Ω–∞—Ü—ñ—î—é —Å—Ç–µ–ø–µ–Ω–µ–≤–∏—Ö —Ñ—É–Ω–∫—Ü—ñ–π
	\begin{equation}\label{star_prb:L}
		f_{nm}(r) = A_{nm}{\left( {\frac{r}{R}} \right)^n} + {B_{nm}}{\left( {\frac{R}{r}} \right)^{n + 1}}
	\end{equation}

	–ù–æ—Ä–º—É–≤–∞–Ω–Ω—è –Ω–∞ $R$ –≤–≤–µ–¥–µ–Ω–æ –¥–ª—è –∑—Ä—É—á–Ω–æ—Å—Ç—ñ, –∫–æ–Ω—Å—Ç–∞–Ω—Ç–∏ $A_{nm}$ —Ç–∞ $B_{nm}$~--- —Ä—ñ–∑–Ω—ñ –≤—Å–µ—Ä–µ–¥–∏–Ω—ñ —ñ –∑–æ–≤–Ω—ñ –∫—É–ª—ñ.
	–ó–æ–≤–Ω—ñ –∫—É–ª—ñ ($r >R$) –ø–æ–ª–µ –æ–±–º–µ–∂–µ–Ω–æ—ó —Å–∏—Å—Ç–µ–º–∏ –∑–∞—Ä—è–¥—ñ–≤ –º–∞—î –ø—Ä—è–º—É–≤–∞—Ç–∏ –¥–æ –Ω—É–ª—è –Ω–∞ –Ω–µ—Å–∫—ñ–Ω—á–µ–Ω–Ω–æ—Å—Ç—ñ, —Ç–æ–º—É $A_{nm} = 0$,

	\begin{equation}\label{dstar_prb:L}
		\pot_\mathrm{out}(r,\theta ,\varphi ) = \sum\limits_{n = 0}^\infty  {\sum\limits_{m =  - n}^n {{B_{nm}}{{\left( {\frac{R}{r}} \right)}^{n + 1}}{Y_{nm}}(\theta ,\varphi )} }.
	\end{equation}

	–ê–Ω–∞–ª–æ–≥—ñ—á–Ω–æ —Ä–æ–∑–≥–ª—è–¥–∞—î–º–æ —Ä–æ–∑–≤‚Äô—è–∑–æ–∫ —É—Å–µ—Ä–µ–¥–∏–Ω—ñ –∫—É–ª—ñ, –¥–µ, –æ–¥–Ω–∞–∫, –ø–æ–ª–µ –º–∞—î –±—É—Ç–∏ —Ä–µ–≥—É–ª—è—Ä–Ω–∏–º –≤ –Ω—É–ª—ñ (\emph{–≤ —Ü–µ–Ω—Ç—Ä—ñ –Ω–µ–º–∞—î —è–∫–∏—Ö–æ—Å—å —Ç–æ—á–∫–æ–≤–∏—Ö –∑–∞—Ä—è–¥—ñ–≤}); —Ç–æ–º—É –≤~\eqref{star_prb:L} —Å–ª—ñ–¥ –≤—ñ–¥–∫–∏–Ω—É—Ç–∏ —á–ª–µ–Ω–∏ –∑ –≤—ñ–¥'—î–º–Ω–∏–º–∏ —Å—Ç–µ–ø–µ–Ω—è–º–∏ –∑–º—ñ–Ω–Ω–æ—ó $r < R$ :
	\begin{equation}\label{ddstar_prb:L}
		\pot_\mathrm{in}(r,\theta ,\varphi ) = \sum\limits_{n = 0}^\infty  {\sum\limits_{m =  - n}^n {{{\tilde A}_{nm}}{{\left( {\frac{r}{R}} \right)}^n}{Y_{nm}}(\theta ,\varphi )} }
	\end{equation}
	–ú–∞—î–º–æ –¥–≤—ñ –≥—Ä–∞–Ω–∏—á–Ω—ñ —É–º–æ–≤–∏ –Ω–∞ –ø–æ–≤–µ—Ä—Ö–Ω—ñ –∫—É–ª—ñ ($r = R$). –£–º–æ–≤—É –¥–ª—è —Ç–∞–Ω–≥–µ–Ω—Ü—ñ–∞–ª—å–Ω–∏—Ö –∫–æ–º–ø–æ–Ω–µ–Ω—Ç –Ω–∞–ø—Ä—É–∂–µ–Ω–æ—Å—Ç—ñ –µ–ª–µ–∫—Ç—Ä–∏—á–Ω–æ–≥–æ –ø–æ–ª—è –≤ –¥–∞–Ω—ñ–π –∑–∞–¥–∞—á—ñ –º–æ–∂–Ω–∞ –∑–∞–º—ñ–Ω–∏—Ç–∏ —É–º–æ–≤–æ—é –Ω–µ–ø–µ—Ä–µ—Ä–≤–Ω–æ—Å—Ç—ñ –ø–æ—Ç–µ–Ω—Ü—ñ–∞–ª—É
	\[
		\pot_\mathrm{out}(R,\theta ,\varphi ) = {\pot_\mathrm{in}}(R,\theta ,\varphi ).
	\]
	–¶–µ –¥–∞—î  $\tilde{A}_{nm} = B_{mn}$.
	–î—Ä—É–≥–∞ –≥—Ä–∞–Ω–∏—á–Ω–∞ —É–º–æ–≤–∞ –≤–∏–∫–æ—Ä–∏—Å—Ç–æ–≤—É—î –Ω–æ—Ä–º–∞–ª—å–Ω—ñ –∫–æ–º–ø–æ–Ω–µ–Ω—Ç–∏ —ñ–Ω–¥—É–∫—Ü—ñ—ó $D_n =  - \epsilon \vect{n} \cdot \vect{\nabla}\pot  =  - \epsilon \frac{\partial \pot }{\partial r}$ , –≤–æ–Ω–∞ –¥–∞—î
	\[
		- \left. {\frac{{\partial {\pot_\mathrm{out}}}}{{\partial r}}} \right|_{r = R} + \varepsilon {\left. {\frac{{\partial {\pot_\mathrm{in}}}}{{\partial r}}} \right|_{r = R}} = 4\pi \sum\limits_{n = 0}^\infty  {\sum\limits_{m =  - n}^n {{s_{nm}}{Y_{nm}}(\theta ,\varphi )} }.
	\]
	–ü—ñ–¥—Å—Ç–∞–Ω–æ–≤–∫–∞~\eqref{dstar_prb:L}, \eqref{ddstar_prb:L} –∑ —É—Ä–∞—Ö—É–≤–∞–Ω–Ω—è–º  $\tilde{A}_{nm} = B_{mn}$ –≤–∏–∑–Ω–∞—á–∞—î —Ü—ñ –∫–æ–µ—Ñ—ñ—Ü—ñ—î–Ω—Ç–∏:
	\[
		\left( n + 1 + n\varepsilon  \right)B_{nm} = 4\pi s_{nm}.
	\]
	–û—Ç–∂–µ, –æ—Å—Ç–∞—Ç–æ—á–Ω–æ –º–∞—î–º–æ:
	\[
		\pot(r,\theta,\phi) =
		\begin{cases}
			4\pi \sum\limits_{n = 0}^\infty  {\sum\limits_{m =  - n}^n {\frac{{{s_{nm}}}}{{n + 1 + n\varepsilon }}{{\left( {\frac{r}{R}} \right)}^n}{Y_{nm}}(\theta ,\varphi )} } , \quad r < R \\
			4\pi \sum\limits_{n = 0}^\infty  {\sum\limits_{m =  - n}^n {\frac{{{s_{nm}}}}{{n + 1 + n\varepsilon }}{{\left( {\frac{R}{r}} \right)}^{n + 1}}{Y_{nm}}(\theta ,\varphi )} } , \quad r \ge R.
		\end{cases}
	\]

\end{solution}
\end{problem}
